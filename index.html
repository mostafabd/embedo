embedo

A simple, fast, lightweight and standalone social media content embedder plugin for web

embedo.js

        
 /**
  * @example Embedo Implementation Sample
  *
  * @package npm install embedo --save
  * @package bower install embedo
  * @import https://www.jsdelivr.com/projects/embedo (via jsdelivr)
  * @import https://cdnjs.com/libraries/embedo (via cdnjs)
  * @import https://unpkg.com/embedo (source via unpkg)
  */
  
  import Embedo from './my/vendor/folder';
  
  const embedo = new Embedo({
    facebook: {
      appId: 'my_app_id',
      version: 'v2.8'
    }, // Enable facebook SDK
    twitter: true,  // Enable twitter SDK
    instagram: true,  // Enable instagram SDK
    pinterest: true  // Enable pinterest SDK
  });

  // Calling .load() event [HTMLElement, url|string, options|object{}]
  embedo.load(
    document.getElementById('my-awesome-container'),
    'https://twitter.com/Sh0bhit/status/830544109857275906'
  );

  // Passing custom options
  embedo.load(
    document.getElementById('my-awesome-container'),
    'https://www.youtube.com/watch?v=5Ad3TH8XGSI',
    {
      width: 640,
      height: 480
    }
  );

  // Also, the jQuery way
  embedo.load($('my-awesome-container').get(0), 'my://url');

  // Refresh event (if no element passed, refreshes all embedo instances)
  embedo.refresh(document.getElementById('my-awesome-container'));

  // Destroy event (if no element passed, destroys all embedo instances)
  embedo.destroy(document.getElementById('my-awesome-container'));

  // Watch Event
  embedo.on('watch', function (result) {
    console.log(result); // Returns element and its dimensions
  });

  // Refresh Event
  embedo.on('refresh', function (request, result) {
    console.log(request, result); // Returns request meta and dimensions
  });

  // Destroy Event
  embedo.on('destroy', function () {
    // Do something when embedo destroys
  });

  // Catch errors during embed, if/any.
  embedo.on('error', function (error) {
    console.error(error); // Emits in case of invalid request or an exception
  });
